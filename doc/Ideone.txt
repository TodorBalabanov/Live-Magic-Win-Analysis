import java.util.*;
import java.lang.*;
import java.io.*;

class Ideone {
	public static void main (String[] args) throws java.lang.Exception {
		long v = 0x7FL;
		long e = 0xFE0000000000L;
		long t = 0;
		int c = 0;
		int o = 0;

		while(v <= e) {
	        	c++;
			if((v & 0x3FL) == 0x3FL) {
				o++;
			}

			t = (v | (v - 1)) + 1;
			v = t | ((((t & -t) / (v & -v)) >> 1) - 1);
		}

		System.out.println( "=" + o + "/" + c );
	}
}
